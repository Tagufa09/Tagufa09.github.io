        using System;

        namespace HelloWorld
        {
          class BookMark
          {
            private string inkColor = "black";
            private int inkContent = 20;
            
            public void display()
            {
              Console.WriteLine("{0} {1}",inkColor,inkContent);
            }
            public void changeC()
            {
              string Ch;
              Console.WriteLine("change color ");
              Ch = Console.ReadLine();
              if(inkContent == 0)
                 {
                   Console.WriteLine("Refill");
                 }
              
              else
                 {
                   inkColor = Ch;
                   Console.WriteLine(inkColor);
                 }              
            }
            public void refill()
            {
              Console.WriteLine("ink content is 20ml");
              inkContent = 20;
            }
            public void write()
            {
              if (inkContent == 0)
              {
                Console.WriteLine("Refill");
              }
              else
              {
                Console.WriteLine("10ml deducted");
                inkContent = inkContent - 10;
              }
            }
          }
                   
          class Program
          {
            static void Main(string[] args)
            {
              BookMark bm = new BookMark();
              int choice;
              do
              {
                Console.WriteLine("---MENU---");
                Console.WriteLine("1. DISPLAY INFO");
                Console.WriteLine("2. CHANGE COLOR");
                Console.WriteLine("3. REFILL");
                Console.WriteLine("4. WRITE");
                Console.WriteLine("5. EXIT");
                
                Console.WriteLine("ENTER YOUR CHOICE");
                choice = Convert.ToInt32(Console.ReadLine());
                
                switch(choice)
                {
                  case 1: bm.display();
                    break;
                  case 2: bm.changeC();
                    break;
                  case 3: bm.refill();
                    break;
                  case 4: bm.write();
                    break;
                  case 5: Console.WriteLine("end");
                          return;
                
                }                                      
              }while(true);
                
            }
          }
        }
